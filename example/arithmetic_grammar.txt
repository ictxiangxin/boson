start: E { $$ = $1 }
     ;
E : E '+' T { $$ = [$2, $1, $3] }
  | E '-' T { $$ = [$2, $1, $3] }
  | T { $$ = $1 }
  ;
T : T '*' F { $$ = [$2, $1, $3] }
  | T '/' F { $$ = [$2, $1, $3] }
  | F { $$ = $1 }
  ;
F : F '^' D { $$ = [$2, $1, $3] }
  | D { $$ = $1 }
  ;
D : '(' E ')' { $$ = [$2] }
  | N { $$ = $1 }
  ;
N : int { $$ = $1 }
  | float { $$ = $1 }
  ;
