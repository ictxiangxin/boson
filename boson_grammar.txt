start : grammar
      ;

grammar : grammar statement
        | statement
        ;

statement : command
          | reduction
          | section
          ;

command : command element_list end
        ;

reduction : name reduce derivation_list end
        ;

section : section_head section_text

derivation_list : derivation_list or derivation
                | derivation
                ;

derivation : name_list code
           | name_list
           ;

element_list : element_list element
             | element
             ;

name_list : name_list name
          | name
          ;

element : name
        | literal
        ;
